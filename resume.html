    class PeterRosal

      attr_reader :task, :institution, :years
      attr_accessor :details

      def initialize(task, institution, years=1)
        @task = task
        @institution = institution
        @years = years
        @details = []
      end 

      def details(actions)
        @details << actions
      end

    end
